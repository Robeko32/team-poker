function calcPagos() {

  var c  = calcPozo();

  var pr = c.active.length ? calcPremios(c.pozo, c.active.length) : {ng:0, amounts:[]};

  var cobroHost = c.active.filter(function(p){return p.name!==T.host;}).length * 8000;

  var bals = c.active.map(function(p) {

    var isHost = p.name === T.host;

    var buyin  = (p.cajasIni + p.rebuys) * cajaVal();

    var isW    = p.pos >= 1 && p.pos <= pr.ng;

    var pmBase = isW ? pr.amounts[p.pos-1] : 0;

    var premio = (p.premioOverride>=0) ? p.premioOverride : pmBase;

    return {name: p.name, bal: isHost ? (premio-buyin+cobroHost) : (premio-buyin-8000)};

  });

  var deb = bals.filter(function(b){return b.bal<0;}).map(function(b){return {name:b.name,bal:-b.bal};}).sort(function(a,b){return b.bal-a.bal;});

  var cre = bals.filter(function(b){return b.bal>0;}).sort(function(a,b){return b.bal-a.bal;});

  var pagos=[]; var di=0, ci=0;

  while(di<deb.length && ci<cre.length) {

    var m = Math.min(deb[di].bal, cre[ci].bal);

    if(m>10) pagos.push({de:deb[di].name, a:cre[ci].name, montoBase:m});

    deb[di].bal -= m; cre[ci].bal -= m;

    if(deb[di].bal<10) di++;

    if(cre[ci].bal<10) ci++;

  }

  return pagos;

}

function commitHistory() {

  if (!T || !T.players) return;

  var c = calcPozo();

  if (c.active.length === 0) return;

  // Prevent committing the same torneo twice — check Firebase directly

  var nroToCommit = T.nro;

  if (nroToCommit === _lastCommittedNro) return;

  _lastCommittedNro = nroToCommit;

  DB.ref("lastCommittedNro").set(nroToCommit);


  var pr = calcPremios(c.pozo, c.active.length);

  c.active.forEach(function(p) {

    if (!H[p.name]) H[p.name] = {veces:0, torneos:0, res:0, prevVeces:0, prevTorneos:0};

    var currentRes = parseFloat(H[p.name].res);

    if (isNaN(currentRes)) currentRes = 0;

    H[p.name].torneos = (parseInt(H[p.name].torneos)||0) + 1;

    if (p.name === T.host) H[p.name].veces = (parseInt(H[p.name].veces)||0) + 1;

    var buyin  = (p.cajasIni+p.rebuys)*cajaVal();

    var isW    = p.pos>=1 && p.pos<=pr.ng;

    var pmBase = isW ? pr.amounts[p.pos-1] : 0;

    var premio = (p.premioOverride >= 0) ? p.premioOverride : pmBase;

    H[p.name].res = currentRes + (premio - buyin);

    if (isW) H[p.name].pts = (parseInt(H[p.name].pts)||0) + (pr.ng - p.pos + 1);

  });

  saveH();


  // Save torneo record to historialTorneos

  var htRecord = {nro:T.nro, fecha:T.fecha, tipo:T.tipo, host:T.host, jugadores:{}};

  c.active.forEach(function(p) {

    var bi2=(p.cajasIni+p.rebuys)*cajaVal();

    var isW2=p.pos>=1&&p.pos<=pr.ng;

    var pm2=(p.premioOverride>=0)?p.premioOverride:(isW2?pr.amounts[p.pos-1]:0);

    htRecord.jugadores[p.name]={cajas:p.cajasIni+p.rebuys,pos:p.pos||0,resultado:pm2-bi2,isHost:p.name===T.host};

  });

  DB.ref("historialTorneos/T"+T.nro).set(htRecord, function(err){ console.log(err ? "historialTorneos ERROR:"+err : "historialTorneos OK T"+T.nro); });

}
